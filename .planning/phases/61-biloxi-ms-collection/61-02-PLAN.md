```bash
cd backend
npx tsx -e "
import { createClient } from '@supabase/supabase-js';
const supabase = createClient(process.env.SUPABASE_URL!, process.env.SUPABASE_SERVICE_ROLE_KEY!);
// Find officeholder questions missing expiresAt
const { data } = await supabase.from('questions')
  .select('id, external_id, question_text, expires_at')
  .like('external_id', 'bxl-%')
  .is('expires_at', null);
console.log('Questions without expiresAt:');
data?.forEach(q => console.log(q.external_id, q.question_text?.substring(0, 80)));
"
```

Review the output. For any questions about current officeholders, set expiresAt:

```bash
cd backend
npx tsx -e "
import { createClient } from '@supabase/supabase-js';
const supabase = createClient(process.env.SUPABASE_URL!, process.env.SUPABASE_SERVICE_ROLE_KEY!);
// Update specific question IDs with expiresAt
const officeholderIds = [/* list IDs from above query */];
const { error } = await supabase.from('questions')
  .update({ expires_at: '2029-06-01T00:00:00Z' })
  .in('id', officeholderIds);

else console.log('Updated', officeholderIds.length, 'questions with expiresAt');
```
