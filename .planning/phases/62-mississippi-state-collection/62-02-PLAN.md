```bash
cd backend
npx tsx -e "
import { createClient } from '@supabase/supabase-js';
const supabase = createClient(process.env.SUPABASE_URL!, process.env.SUPABASE_SERVICE_ROLE_KEY!);
const { data } = await supabase.from('questions')
  .select('id, external_id, question_text, expires_at, correct_answer')
  .like('external_id', 'mis-%')
  .eq('status', 'active');
const officeholderKeywords = ['governor', 'attorney general', 'secretary of state', 'treasurer', 'speaker', 'reeves', 'hosemann', 'fitch', 'watson', 'mcrae', 'jason white', 'lamar'];
const officeholderQs = data?.filter(q =>
  officeholderKeywords.some(k =>
    q.question_text.toLowerCase().includes(k) || q.correct_answer?.toLowerCase().includes(k)
  )
);
console.log('Officeholder questions found:', officeholderQs?.length);
officeholderQs?.forEach(q => console.log(q.external_id, q.expires_at ? 'HAS expiry' : 'MISSING expiry', q.question_text.substring(0, 80)));
"
```

Update all officeholder questions to have `expiresAt: '2028-01-13T00:00:00Z'`:

```bash
cd backend
npx tsx -e "
import { createClient } from '@supabase/supabase-js';
const supabase = createClient(process.env.SUPABASE_URL!, process.env.SUPABASE_SERVICE_ROLE_KEY!);
const { data } = await supabase.from('questions')
  .select('id, external_id, question_text, correct_answer')
  .like('external_id', 'mis-%')
  .is('expires_at', null);
const officeholderKeywords = ['governor', 'attorney general', 'secretary of state', 'treasurer', 'speaker', 'reeves', 'hosemann', 'fitch', 'watson', 'mcrae', 'jason white', 'lamar'];
const toUpdate = data?.filter(q =>
  officeholderKeywords.some(k =>
    q.question_text.toLowerCase().includes(k) || q.correct_answer?.toLowerCase().includes(k)
  )
) || [];
console.log('Updating', toUpdate.length, 'questions with expiresAt');
for (const q of toUpdate) {
  const { error } = await supabase.from('questions')
    .update({ expires_at: '2028-01-13T00:00:00Z' })
    .eq('id', q.id);
  if (error) console.error('Failed:', q.external_id, error);
  else console.log('Updated:', q.external_id);
}
"
```
