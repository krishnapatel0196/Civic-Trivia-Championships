    The cleanest approach is a small async function that both ClusterBuilder and CollectionHierarchy can call. Add it to types.ts or create a new utility. Recommendation: add to types.ts since it already exports the tier types:

    ```typescript
    /**
     * Load collection tier map from database at runtime.
     * Returns Map<collectionName, CollectionTier> keyed by display name
     * (matching the previous COLLECTION_HIERARCHY key format).
     */
    export async function loadCollectionTierMap(): Promise<Map<string, CollectionTier>> {
      const { db } = await import('../../db/index.js');
      const { collections } = await import('../../db/schema.js');
      const rows = await db.select({ name: collections.name, tier: collections.tier }).from(collections);
      const map = new Map<string, CollectionTier>();
      for (const row of rows) {
        map.set(row.name, row.tier as CollectionTier);
      }
      return map;
    }
    ```
